public class Main{
    public static int[] SelectionSort(int[] inputArr){     
        int arrSize = inputArr.length;
        int temp = 0;
        for(int i = 0; i<arrSize-1;i++){
            int smallestInd = i;
            for(int j=i; j<arrSize;j++){
                if(inputArr[j]<inputArr[smallestInd]){
                    smallestInd = j;
                }
            }
            if(smallestInd!=i){
                temp = inputArr[i];
                inputArr[i] = inputArr[smallestInd];
                inputArr[smallestInd]=temp;

            }
        }
        return inputArr;
    }
    public static void main(String[] args) {
        int[] testArr = {111,5,6,5,43};
        System.out.println(testArr);
        int[] result = SelectionSort(testArr);
        System.out.println("result: ");
        for(int num:result){
            System.out.print(num+" ");
        }
    }
}
