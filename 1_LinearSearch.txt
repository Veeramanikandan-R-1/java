public class LinearSearch{
    public int findLinearSearch(int[] inputArr, int targetElement){
        for(int i =0; i<inputArr.length;i++){
            if(inputArr[i]==targetElement){
                return 1;
            }
        }
        return -1;
    }
    public static void main(String[] args) {
        int[] testArr = {1,5,6,5,43};
        System.out.println(testArr);
        LinearSearch mainObj = new LinearSearch();
        int result = mainObj.findLinearSearch(testArr, 15);
        System.out.println("result: "+result);
    }
}
