public class Main{
    public int BinarySearch(int[] inputArr, int targetElement){
        int left = 0;
        int right = inputArr.length-1;

        while(left <= right){
            int mid = left + (right-left)/2;
            int midElmnt = inputArr[mid];
            if(midElmnt==targetElement){
                return  1;
            }else{
                if(midElmnt<targetElement){
                    left = mid+1;
                }else{
                    right = mid-1;
                }
            }
        }
        return -1;

    }
    public static void main(String[] args) {
        int[] testArr = {1,5,6,5,43};
        System.out.println(testArr);
        Main mainObj = new Main();
        int result = mainObj.BinarySearch(testArr, 5);
        System.out.println("result: "+result);
    }
}
